187 #if !defined(MBEDTLS_ECP_WINDOW_SIZE)
188 /*
189  * Maximum "window" size used for point multiplication.
190  * Default: 6.
191  * Minimum value: 2. Maximum value: 7.
192  *
193  * Result is an array of at most ( 1 << ( MBEDTLS_ECP_WINDOW_SIZE - 1 ) )
194  * points used for point multiplication. This value is directly tied to EC
195  * peak memory usage, so decreasing it by one should roughly cut memory usage
196  * by two (if large curves are in use).
197  *
198  * Reduction in size may reduce speed, but larger curves are impacted first.
199  * Sample performances (in ECDHE handshakes/s, with FIXED_POINT_OPTIM = 1):
200  * w-size: 6 5 4 3 2
201  * 521 145 141 135 120 97
202  * 384 214 209 198 177 146
203  * 256 320 320 303 262 226
204 
205  * 224 475 475 453 398 342
206  * 192 640 640 633 587 476
207  */
208 #define MBEDTLS_ECP_WINDOW_SIZE 6
209 #endif /* MBEDTLS_ECP_WINDOW_SIZE */
210 
211 #if !defined(MBEDTLS_ECP_FIXED_POINT_OPTIM)
212 /*
213  * Trade memory for speed on fixed-point multiplication.
214  *
215  * This speeds up repeated multiplication of the generator (that is, the
216  * multiplication in ECDSA signatures, and half of the multiplications in
217  * ECDSA verification and ECDHE) by a factor roughly 3 to 4.
218  *
219  * The cost is increasing EC peak memory usage by a factor roughly 2.
220  *
221  * Change this value to 0 to reduce peak memory usage.
222  */
223 #define MBEDTLS_ECP_FIXED_POINT_OPTIM 1
224 #endif /* MBEDTLS_ECP_FIXED_POINT_OPTIM */
