1051  }

1123  break;
1124  }
1125  }
1126  else
1127  {
1128  break;
1129  }
1130  }
1131  else
1132  {
1133  break;
1134  }
1135 
1136  gpdataOffset += toRead;
1137  bytesToRead -= toRead;
1138  outOffset += toRead;
1139  }
1140 
1141  FreeAPDUBuffer(pApdu);
1142 
1143  return rv;
1144 }
