120 int ax_mbedtls_associate_keypair(
121  SST_Index_t key_index, mbedtls_pk_context * pkey)
122 {
123  mbedtls_ecp_keypair * pax_ctx;
124  HLSE_OBJECT_HANDLE keyPairHandles[A71CH_KEY_PAIR_MAX];
125  U16 kpHandleNum = A71CH_KEY_PAIR_MAX;
126  U16 kpHandleNumMax;
127  U8 i = 0;
128  HLSE_RET_CODE err;
129 
130  if (key_index >= (sizeof(gmbedtls_ecp_keypair)/sizeof(gmbedtls_ecp_keypair[0]))) {
132  return 1;
133  }
134  pkey->pk_info = &ax_mbedtls_eckeypair_info;
135  if (gmbedtls_ecp_keypair[key_index] == NULL)
136  {
137  gmbedtls_ecp_keypair[key_index] = (mbedtls_ecp_keypair *) mbedtls_calloc(1, sizeof(mbedtls_ecp_keypair));
138  }
139  pax_ctx = gmbedtls_ecp_keypair[key_index];
140  err = HLSE_EnumerateObjects(HLSE_KEY_PAIR, keyPairHandles, &kpHandleNum);
141  kpHandleNumMax = kpHandleNum;
142  if (err != HLSE_SW_OK)
143  {
144  return 1;
145  }
146  while(kpHandleNum)
147  {
148  if (HLSE_GET_OBJECT_INDEX(keyPairHandles[i]) == key_index)
149  {
150  pax_ctx->grp.hlse_handle = keyPairHandles[i];
151  break;
152  }
153  i++;
154  kpHandleNum--;
155  }
156  if (i >= kpHandleNumMax)
157  {
158  /*unable to find the keypair at specified index */
159  return 1;
160  }
161  pax_ctx->grp.id = MBEDTLS_ECP_DP_SECP256R1;
162 #if LOG_API_CALLS
163  sm_printf(DBGOUT, "Associating ECC key-pair '%d'.\r\n",HLSE_GET_OBJECT_INDEX(pax_ctx->grp.hlse_handle));
164 #endif /* LOG_API_CALLS */
165  pkey->pk_ctx = pax_ctx;
166  return 0;
167 }
