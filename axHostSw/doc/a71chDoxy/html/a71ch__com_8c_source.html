65 {
66  // the IP address is in format a.b.c.d:port, e.g. 10.0.0.1:8080
67  int nSuccess;
68 
69  if (strlen(arg) >= szServerLen) {
70  return ERR_BUF_TOO_SMALL;
71  }
72 
73  // First attempt at parsing: server IP-address passed, sscanf will return 2 upon successfull parsing
74  nSuccess = sscanf(arg, "%15[0-9.]:%5u[0-9]", szServer, (unsigned int *)port);
75  if (nSuccess == 2) {
76  return SW_OK;
77  }
78  else {
79  // Second attempt at parsing: server name passed instead of IP-address
80  unsigned int i;
81  int fColonFound = 0;
82 
83  for (i = 0; i < strlen(arg); i++) {
84  if (arg[i] == ':') {
85  szServer[i] = 0;
86  fColonFound = 1;
87  // PRINTF("servername: %s\r\n", szServer);
88  break;
89  }
90  else {
91  szServer[i] = arg[i];
92  }
93  }
94 
95  if ((fColonFound == 1) && (i != 0)) {
96  nSuccess = sscanf(&arg[i], ":%5u[0-9]", (unsigned int *)port);

98  return SW_OK;
99  }
100  }
101  }
102  return ERR_NO_VALID_IP_PORT_PATTERN;
103 }
