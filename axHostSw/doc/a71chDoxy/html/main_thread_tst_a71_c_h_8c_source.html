58 int main(int argc, char ** argv)
59 {
60  pthread_t tId[nThread];
61  int testStatus[nThread];
62  int nError;
63  int *pExitCode;
64  int i;
65  int fOverallResult = TEST_SUCCESS;
66 
67  sm_printf(DBGOUT, "A71CH Multi-threaded HostLibrary test application (Rev %s)\n", TST_APP_VERSION);
68  sm_printf(DBGOUT, "**********************************************\n");
69 
70  if (argc != 1)
71  {
72  sm_printf(DBGOUT, "No command line arguments supported.\n");
73  return 20;
74  }
75 
76  for (i=0; i<nThread; i++)
77  {
78  // PRINTF("LINE: %d.", __LINE__);
79  testStatus[i] = TEST_FAILURE;
80  nError = pthread_create(&(tId[i]), NULL, tstTaskA71CH, &(testStatus[i]));
81  if (nError != 0)
82  {
83  PRINTF("Failed to create thread: %s.\n", strerror(nError));
84  return 21;
85  }
86  else
87  {
88  PRINTF("Created thread %d (id=0x%08X).\n", i, tId[i]);
89  }
90  }
91 
92  for (i=0; i<nThread; i++)
93  {
94  nError = pthread_join(tId[i], (void *)&pExitCode);
95  if (nError == 0)
96  {
97  PRINTF("Joined thread %d (id=0x%08X) successfully; Exit code = %d.\n", i, tId[i], *pExitCode);
98  }
99  else
100  {
101  PRINTF("Failed to join thread: %s (thread already terminated?).\n", strerror(nError));
102  return 21;
103  }
104  }
105 
106  for (i=0; i<nThread; i++)
107  {
108  if (testStatus[i] == TEST_SUCCESS)
109  {
110  PRINTF("Test executed in thread %d (id=0x%08X) successfull.\n", i, tId[i]);
111  }
112  else
113  {
114  PRINTF("Test executed in thread %d (id=0x%08X) failed with Test Status: %d.\n", i, tId[i], testStatus[i]);
115  fOverallResult = TEST_FAILURE;
116  }
117  }
118 
119  if (fOverallResult == TEST_SUCCESS)
120  {
121  PRINTF("Overall test succeeded.\n");
122  }
123  else
124  {
125  PRINTF("Overall test failed.\n");
126  }
127 
128  return 0;
129 }
