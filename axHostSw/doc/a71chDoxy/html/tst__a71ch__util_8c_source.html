145  else if (initMode == INIT_MODE_RESET)

147  return "Reset";

181 {

184 #else

191  U8 keyEnc[SCP_KEY_SIZE];

193  U8 keyDek[SCP_KEY_SIZE];

196 
197  PRINTF( "\n-----------\nStart a71chSetupScp03()\n------------\n");
198 
199  DEV_ClearChannelState();
200 
201  // Security module generates random data for initial SCP03 keys
202  sm_printf(CONSOLE, "\nA71_GetRandom(randomLen=%d)\n", randomLen);
203  err = A71_GetRandom(random, randomLen);
204  result &= AX_CHECK_SW(err, SW_OK, "err");
205  if (result != 1) { goto SCP03_EXIT; }
206 
207  // Storing Static Keys
208  memcpy(keyEnc, random, SCP_KEY_SIZE);
209  memcpy(keyMac, random + SCP_KEY_SIZE, SCP_KEY_SIZE);
210  memcpy(keyDek, random + (2*SCP_KEY_SIZE), SCP_KEY_SIZE);
211 
212  keyVersion = (U8) (SST_HOST_SCP_KEYSET >> 8);
213  sm_printf(CONSOLE, "\nSCP_GP_PutKeys(keyVersion=0x%02X)\n", keyVersion);
214  err = SCP_GP_PutKeys(keyVersion, keyEnc, keyMac, keyDek, currentKeyDek, AES_KEY_LEN_nBYTE);
215  result &= AX_CHECK_SW(err, SW_OK, "err");
216  if (result != 1) { goto SCP03_EXIT; }
217 
218  // Authenticate Channel
219  sm_printf(CONSOLE, "\nSCP_Authenticate()\n");
220  err = SCP_Authenticate(keyEnc, keyMac, keyDek, SCP_KEY_SIZE, sCounter, &sCounterLen);
221  result &= AX_CHECK_SW(err, SW_OK, "err");
222  result &= AX_CHECK_U16(sCounterLen, 0, "Only expected when SCP03 is configured for pseudo-random challenge");
223 
224 SCP03_EXIT:
225  PRINTF( "\n-----------\nEnd a71chSetupScp03(), result = %s\n------------\n",
226  ((result == 1)? "OK": "FAILED"));
227 #endif
228  return result;
229 }
