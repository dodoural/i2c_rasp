494 U16 A71_GetSha256(U8 *data, U16 dataLen, U8 *sha, U16 *shaLen)
495 {
496  U8 isOk = 0;
497  U16 rv;
498  U16 maxChunk = 0;
499  U16 remainingData = dataLen;
500  U16 toSend = 0;
501  U16 dataOffset = 0;
502 
503  apdu_t apdu;
504  apdu_t * pApdu = (apdu_t *) &apdu;
505 
506  assert(data != NULL);
507  assert(sha != NULL);
508 
509  pApdu->cla = AX_CLA;
510  pApdu->ins = A71CH_INS_MODULE_GET_SHA256;
511  pApdu->p1 = 0x00;
512  pApdu->p2 = P2_SHA256_ONE_SHOT;
513 
514  AllocateAPDUBuffer(pApdu);
515 
516  maxChunk = A71CH_SHA256_MAX_DATA_CHUNK;
517 
518  // Depending on dataLen we can do a one-shot SHA256 or need to split up across multiple APDU's
519  if (dataLen <= maxChunk)
520  {
521  SetApduHeader(pApdu, USE_STANDARD_APDU_LEN);
522  smApduAppendCmdData(pApdu, data, dataLen);
523 
524  rv = (U16)scp_Transceive(pApdu, SCP_MODE);
525  if (rv == SMCOM_OK)
526  {
527  rv = smGetSw(pApdu, &isOk);
528  if (isOk)
529  {
530  rv = smApduGetResponseBody(pApdu, sha, shaLen);
531  if (rv == SW_OK)
532  {
533  if (32 != *shaLen)
534  {
535  rv = ERR_WRONG_RESPONSE;
536  }
537  }
538  }
539  }
540  }
541  else
542  {
543  // Split up in multiple transactions ...
544  // Send out first chunk (SHA256_Init)
545  pApdu->p2 = P2_SHA256_INIT;
546  SetApduHeader(pApdu, USE_STANDARD_APDU_LEN);
547  toSend = maxChunk;
548  remainingData -= toSend;
549  smApduAppendCmdData(pApdu, data, toSend);
550  dataOffset += toSend;
551 
552  rv = (U16)scp_Transceive(pApdu, SCP_MODE);
553  if (rv == SMCOM_OK)
554  {
555  if (rv == SMCOM_OK)
556  {
557  // No response data expected
558  rv = CheckNoResponseData(pApdu);
559  if (rv != SW_OK) { goto LBL_LEAVE_A71_GetSha256; }
560  }
561  }
562 
563  while (remainingData > maxChunk)
564  {
565  // Send out 'middle' chunks (SHA256_Update)
566  pApdu->p2 = P2_SHA256_UPDATE;
567  SetApduHeader(pApdu, USE_STANDARD_APDU_LEN);
568  toSend = maxChunk;
569  remainingData -= toSend;
570  smApduAppendCmdData(pApdu, &data[dataOffset], toSend);
571  dataOffset += toSend;
572 
573  rv = (U16)scp_Transceive(pApdu, SCP_MODE);
574  if (rv == SMCOM_OK)
575  {
576  if (rv == SMCOM_OK)
577  {
578  // No response data expected
579  rv = CheckNoResponseData(pApdu);
580  if (rv != SW_OK) { goto LBL_LEAVE_A71_GetSha256; }
581  }
582  }
583  }
584 
585  // Always close with SHA256_Final
586  pApdu->p2 = P2_SHA256_FINAL;
587  SetApduHeader(pApdu, USE_STANDARD_APDU_LEN);
588  toSend = remainingData;
589  // remainingData -= toSend;
590  // dataOffset += toSend;
591  smApduAppendCmdData(pApdu, &data[dataOffset], toSend);
592 
593  rv = (U16)scp_Transceive(pApdu, SCP_MODE);
594  {
595  rv = smGetSw(pApdu, &isOk);
596  if (isOk)
597  {
598  rv = smApduGetResponseBody(pApdu, sha, shaLen);
599  if (rv == SW_OK)
600  {
601  if (32 != *shaLen)
602  {
603  rv = ERR_WRONG_RESPONSE;
604  }
605  }
606  }
607  }
608  }
609 
610 LBL_LEAVE_A71_GetSha256:
611  FreeAPDUBuffer(pApdu);
612  return rv;
613 }
