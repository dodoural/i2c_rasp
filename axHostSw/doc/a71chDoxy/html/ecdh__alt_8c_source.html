52 /*
53  * Compute shared secret (SEC1 3.3.1)
54  */
55 int mbedtls_ecdh_compute_shared_o( mbedtls_ecp_group *grp, mbedtls_mpi *z,
56  const mbedtls_ecp_point *Q, const mbedtls_mpi *d,
57  int (*f_rng)(void *, unsigned char *, size_t),
58  void *p_rng )
59 {
60  int ret;
61  mbedtls_ecp_point P;
62 
63  mbedtls_ecp_point_init( &P );
64 
65  /*
66  * Make sure Q is a valid pubkey before using it
67  */
68  MBEDTLS_MPI_CHK( mbedtls_ecp_check_pubkey( grp, Q ) );
69 
70  MBEDTLS_MPI_CHK( mbedtls_ecp_mul( grp, &P, d, Q, f_rng, p_rng ) );
71 
72  if( mbedtls_ecp_is_zero( &P ) )
73  {
74  ret = MBEDTLS_ERR_ECP_BAD_INPUT_DATA;
75  goto cleanup;
76  }
77 
78  MBEDTLS_MPI_CHK( mbedtls_mpi_copy( z, &P.X ) );
79 
80 cleanup:
81  mbedtls_ecp_point_free( &P );
82 
83  return( ret );
84 }

112 /*
113  * Setup and write the ServerKeyExhange parameters (RFC 4492)
114  * struct {
115  * ECParameters curve_params;
116  * ECPoint public;
117  * } ServerECDHParams;
118  */
119 int mbedtls_ecdh_make_params( mbedtls_ecdh_context *ctx, size_t *olen,
120  unsigned char *buf, size_t blen,
121  int (*f_rng)(void *, unsigned char *, size_t),
122  void *p_rng )
123 {
124  int ret;
125  size_t grp_len, pt_len;
126 
127  if( ctx == NULL || ctx->grp.pbits == 0 )
128  return( MBEDTLS_ERR_ECP_BAD_INPUT_DATA );
129 
130  if( ( ret = mbedtls_ecdh_gen_public( &ctx->grp, &ctx->d, &ctx->Q, f_rng, p_rng ) )
131  != 0 )
132  return( ret );
133 
134  if( ( ret = mbedtls_ecp_tls_write_group( &ctx->grp, &grp_len, buf, blen ) )
135  != 0 )
136  return( ret );
137 
138  buf += grp_len;
139  blen -= grp_len;
140 
141  if( ( ret = mbedtls_ecp_tls_write_point( &ctx->grp, &ctx->Q, ctx->point_format,
142  &pt_len, buf, blen ) ) != 0 )
143  return( ret );
144 
145  *olen = grp_len + pt_len;
146  return( 0 );
147 }

149 /*
150  * Read the ServerKeyExhange parameters (RFC 4492)
151  * struct {
152  * ECParameters curve_params;
153  * ECPoint public;
154  * } ServerECDHParams;
155  */
156 int mbedtls_ecdh_read_params( mbedtls_ecdh_context *ctx,
157  const unsigned char **buf, const unsigned char *end )
158 {
159  int ret;
160 
161  if( ( ret = mbedtls_ecp_tls_read_group( &ctx->grp, buf, end - *buf ) ) != 0 )
162  return( ret );
163 
164  if( ( ret = mbedtls_ecp_tls_read_point( &ctx->grp, &ctx->Qp, buf, end - *buf ) )
165  != 0 )
166  return( ret );
167 
168  return( 0 );
169 }

171 /*
172  * Get parameters from a keypair
173  */
174 int mbedtls_ecdh_get_params_o( mbedtls_ecdh_context *ctx, const mbedtls_ecp_keypair *key,
175  mbedtls_ecdh_side side )
176 {
177  int ret;
178  if( ( ret = mbedtls_ecp_group_copy( &ctx->grp, &key->grp ) ) != 0 )
179  return( ret );
180 
181  /* If it's not our key, just import the public part as Qp */
182  if( side == MBEDTLS_ECDH_THEIRS )
183  return( mbedtls_ecp_copy( &ctx->Qp, &key->Q ) );
184 
185  /* Our key: import public (as Q) and private parts */
186  if( side != MBEDTLS_ECDH_OURS )
187  return( MBEDTLS_ERR_ECP_BAD_INPUT_DATA );
188 
189  if( ( ret = mbedtls_ecp_copy( &ctx->Q, &key->Q ) ) != 0 ||
190  ( ret = mbedtls_mpi_copy( &ctx->d, &key->d ) ) != 0 )
191  return( ret );
192 
193  return( 0 );
194 }

221  const unsigned char *buf, size_t blen )
222 {

238 /*
239  * Derive and export the shared secret
240  */
241 int mbedtls_ecdh_calc_secret( mbedtls_ecdh_context *ctx, size_t *olen,
242  unsigned char *buf, size_t blen,
243  int (*f_rng)(void *, unsigned char *, size_t),
244  void *p_rng )
245 {
246  int ret;
247 
248  if( ctx == NULL )
249  return( MBEDTLS_ERR_ECP_BAD_INPUT_DATA );
250 
251  if( ( ret = mbedtls_ecdh_compute_shared( &ctx->grp, &ctx->z, &ctx->Qp, &ctx->d,
252  f_rng, p_rng ) ) != 0 )
253  {
254  return( ret );
255  }
256 
257  if( mbedtls_mpi_size( &ctx->z ) > blen )
258  return( MBEDTLS_ERR_ECP_BAD_INPUT_DATA );
259 
260  *olen = ctx->grp.pbits / 8 + ( ( ctx->grp.pbits % 8 ) != 0 );
261  return mbedtls_mpi_write_binary( &ctx->z, buf, *olen );
262 }
