382 {

385  int foundTag = 0;
386  U16 bufferLen = *pLen;
387 
388  *pLen = 0;
389 
390  if (pApdu->rxlen < 2) /* minimum: 2 byte for response */
391  {
392  return ERR_GENERAL_ERROR;
393  }
394  else
395  {
396  /* check status returned is okay */
397  if ((pApdu->pBuf[pApdu->rxlen - 2] != 0x90) || (pApdu->pBuf[pApdu->rxlen - 1] != 0x00))
398  {
399  return ERR_GENERAL_ERROR;
400  }
401  else // response okay
402  {
403  pApdu->offset = 0;
404 
405  do
406  {
407  U16 len = 0;
408 
409  // Ensure we don't parse beyond the APDU Response Data
410  if (pApdu->offset >= (pApdu->rxlen -2)) { break; }
411 
412  /* get the tag (see ISO 7816-4 annex D); limited to max 2 bytes */
413  if ((pApdu->pBuf[pApdu->offset] & 0x1F) != 0x1F) /* 1 byte tag only */
414  {
415  tag = (pApdu->pBuf[pApdu->offset] & 0x00FF);
416  pApdu->offset += 1;
417  }
418  else /* tag consists out of 2 bytes */
419  {
420  tag = (pApdu->pBuf[pApdu->offset] << 8) + pApdu->pBuf[pApdu->offset + 1];
421  pApdu->offset += 2;
422  }
423 
424  // Ensure we don't parse beyond the APDU Response Data
425  if (pApdu->offset >= (pApdu->rxlen -2)) { break; }
426 
427  // tag is OK
428  /* get the length (see ISO 7816-4 annex D) */
429  if ((pApdu->pBuf[pApdu->offset] & 0x80) != 0x80)
430  {
431  /* 1 byte length */
432  len = (pApdu->pBuf[pApdu->offset++] & 0x00FF);
433  }
434  else
435  {
436  /* length consists of 2 or 3 bytes */
437 
438  U8 additionalBytesForLength = (pApdu->pBuf[pApdu->offset++] & 0x7F);
439

442  len = pApdu->pBuf[pApdu->offset];
443  pApdu->offset += 1;
444  }
445  else if (additionalBytesForLength == 2)
446  {

448  pApdu->offset += 2;
449  }
450  else
451  {
452  return ERR_GENERAL_ERROR;
453  }
454  }
455 
456  // Ensure we don't parse beyond the APDU Response Data
457  if (pApdu->offset >= (pApdu->rxlen -2)) { break; }
458
