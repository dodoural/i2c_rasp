421  }
422  return lReturn;
423  }
424

426  gMappingTableRead = 1;
427 
428  return lReturn;
429 }

765 static HLSE_RET_CODE GetOffsetForNewObjectAndAddNewEntry(U16 newObjectSize, HLSE_GP_DATA_TABLE_ENTRY* entry)
766 {
767  HLSE_RET_CODE lReturn = HLSE_SW_OK;
768  HLSE_GP_DATA_TABLE table;
769  U16 gpSize;
770  U16 newObjectSizeInChunks;
771  U8 found = 0;
772  U8 entryNum;
773  U8 bEntryAlreadyExists = 0;
774 
775  // required for retrieving cerdtials map to know if a segment is locked
776  U16 rv = 0;
777  U8 map[A71CH_MAP_SIZE_MAX];
778  U16 mapLen = sizeof(map);
779 
780  lReturn = ReadGPDataTable(&table, 0);
781  if (lReturn != HLSE_SW_OK)
782  return lReturn;
783 
784  // Check if data object already exists - if so return an error
785  for (entryNum = 0; entryNum < table.numOfEntries; ++entryNum) {
786  if (table.entries[entryNum].index == entry->index && table.entries[entryNum].klass == entry->klass) {
787  bEntryAlreadyExists = 1;
788  break;
789  }
790  }
791  if (bEntryAlreadyExists) {
792  return HLSE_OBJ_ALREADY_EXISTS; // entry already exists
793  }
794 
795  // resolve INDIRECT lengths
796  for (entryNum = 0; entryNum < table.numOfEntries; ++entryNum) {
797  if (table.entries[entryNum].index != 0xFF) {
798  // INDIRECT length and no real length found yet
799  if (table.entries[entryNum].length == HLSE_INDIRECT_LENGTH) {
800  // get the length from the TLV data
801  U8 header[6];
802  U16 actualLength = 0;
803 
804  lReturn = A71_GetGpData(table.entries[entryNum].offset, header, 6);
805  if (lReturn != HLSE_SW_OK)
806  return lReturn;
807 
808  actualLength = GetFileSize(header);
809  table.entries[entryNum].length = HLSE_INDIRECT_LENGTH | actualLength;
810  }
811  }
812  }
813 
814  lReturn = GetGPDataSize(&gpSize);
815  if (lReturn != HLSE_SW_OK)
816  return lReturn;
817 
818  // obtain the credentials info map, to be able to check for locked segments
819  rv = A71_GetCredentialInfo(map, &mapLen);
820  if (rv != SW_OK) {
821  return HLSE_ERR_API_ERROR;
822  }
823 
824  // now find a unoccupied (and unlocked) large enough offset for the new object, assuming the table is sorted in ascending order of offsets
825  newObjectSizeInChunks = HLSE_ALIGN_SIZE(newObjectSize);
826 
827  entry->offset = 0;
828 
829  // invalid entries will have length = 0xFFFF so they will be at the end
830 
831  // check if it possible to insert the object at the begining of the GP Storage
832  if (table.numOfEntries == 0 || (table.entries[0].offset / HLSE_GP_DATA_CHUNK) >= newObjectSizeInChunks) {
833  if (MemoryIsUnlocked(0, newObjectSize, map, mapLen)) {
834  entry->offset = 0;
835  found = 1;
836  }
837  }
838  else {
839  U8 leftEntry, rightEntry;
840  for (leftEntry = 0; leftEntry < table.numOfEntries && !found; ++leftEntry) {
841  if (table.entries[leftEntry].index != 0xFF && table.entries[leftEntry].klass != 0xFF) {
842  // leftEntry is a valid entry
843  rightEntry = leftEntry + 1;
844  if (rightEntry < table.numOfEntries && table.entries[rightEntry].index != 0xFF && table.entries[rightEntry].klass != 0xFF) {
845  // rightEntry is a valid entry
846  U16 leftEnd = HLSE_ALIGN_SIZE(table.entries[leftEntry].offset + (table.entries[leftEntry].length & 0x7FFF));
847  // Check the validiy of candidate chunks from leftEnd to beginning of 'rightEntry' object
848  while (!found && (((table.entries[rightEntry].offset / HLSE_GP_DATA_CHUNK) - leftEnd) >= newObjectSizeInChunks)) {
849  // Verify memory area is unlocked
850  if (MemoryIsUnlocked(leftEnd * HLSE_GP_DATA_CHUNK, newObjectSize, map, mapLen)) {
851  entry->offset = leftEnd;
852  found = 1;
853  }
854  else {
855  // skip to next chunk to begin check
856  leftEnd++;
857  }
858  }
859  }
860  }
861  }
862 
863  if (!found) {
864  // check if there is space in the end
865 
866  // find the last valid entry
867  S8 lastValidEntry = table.numOfEntries - 1;
868  U8 nCurGpTableChunks;
869  for (; lastValidEntry >= 0; --lastValidEntry) {
870  if (table.entries[lastValidEntry].index != 0xFF)
871  break;
872  }
873 
874  // no valid entries - check that the gpData is big enough (excluding last chunk(s) of gp table)
875  nCurGpTableChunks = HLSE_ALIGN_SIZE(((table.numOfEntries * 6) + 2));
876  if (lastValidEntry == 0 && table.entries[lastValidEntry].index == 0xFF && (HLSE_ALIGN_SIZE(gpSize) - nCurGpTableChunks) >= newObjectSizeInChunks) {
877  if (MemoryIsUnlocked(0, newObjectSize, map, mapLen)) {
878  entry->offset = 0;
879  found = 1;
880  }
881  }
882  else {
883  // check for valid free memory area after the last object
884  U16 prevObjectEnd = HLSE_ALIGN_SIZE(table.entries[lastValidEntry].offset + (table.entries[lastValidEntry].length & 0x7FFF));
885  while (!found && ((HLSE_ALIGN_SIZE(gpSize) - nCurGpTableChunks - prevObjectEnd) >= newObjectSizeInChunks)) {
886  if (MemoryIsUnlocked(prevObjectEnd * HLSE_GP_DATA_CHUNK, newObjectSize, map, mapLen)) {
887  entry->offset = prevObjectEnd;
888  found = 1;
889  }
890  else {
891  // skip to next chunk to begin check
892  prevObjectEnd++;
893  }
894  }
895  }
896  }
897  }
898 
899  if (found) {
900  entry->offset *= HLSE_GP_DATA_CHUNK;
901  }
902  else {
903  entry->offset = 0;
904  return HLSE_ERR_MEMORY;
905  }
906 
907  // check if there is an empty entry
908  found = 0;
909  for (entryNum = 0; entryNum < table.numOfEntries; ++entryNum) {
910  if (table.entries[entryNum].index == 0xFF) {
911  memcpy(&table.entries[entryNum], entry, sizeof(HLSE_GP_DATA_TABLE_ENTRY));
912  found = 1;
913  break;
914  }
915  }
916 
917  if (!found) {
918  table.numOfEntries++;
919  if (table.numOfEntries > HLSE_MAX_OBJECTS_IN_TABLE) {
920  return HLSE_ERR_IDENT_IDX_RANGE;
921  }
922  memcpy(&table.entries[table.numOfEntries - 1], entry, sizeof(HLSE_GP_DATA_TABLE_ENTRY));
923  }
924 
925  // sort the entries in ascending order of the offset
926  SortTable(&table);
927 
928  lReturn = WriteGPDataTable(&table);
929 
930  return lReturn;
931 }
