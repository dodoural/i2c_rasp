173  U8 apduPayloadToEncrypt[SCP_BUFFER_SIZE];
174 
175  Scp03SessionState_t session;

177  // Get the current security level. (should be set to AUTH_C_DECRYPTION_R_ENCRYPTION in the GP_ExternalAuthenticate command)
178 
179  /* The Off-Card Entity first encrypts the Command Data field and then computes the C-MAC on the
180  command with the ciphered data field as described in section 6.2.4 APDU Command C-MAC Generation
181  and Verification. No encryption shall be applied to a command where there is no command data field. */
182 
183  // Patch CLA byte to indicate we are using SCP03
184  pApdu->pBuf[0] |= 0x04;
185 
186  if (pApdu->hasData == 1)
187  {
188  int payloadOffset = 0;
189 
190  /* Prior to encrypting the data, the data shall be padded as defined in section 4.1.4. This padding becomes part of the data field.*/
191  // DPRINTF("pApdu->lc: %d\r\n", pApdu->lc);
192  Lcc = pApdu->lc;
193  Lcc += scp_PadData(pApdu);
194  // DPRINTF("Lcc: %d (@line %d)\r\n", Lcc, __LINE__);
195 
196  /* The final Lc value (Lcc) is the sum of: initial Lc + length of the padding + length of C-MAC */
197  Lcc += CMAC_SIZE;
198  // DPRINTF("Lcc: %d (@line %d)\r\n", Lcc, __LINE__);

201

203  if (nRet == HOST_CRYPTO_OK)
204  {

206

210  }
211  else
212  {
213  payloadOffset = APDU_OFFSET_LC + 1; // (1 byte reserved for LC field)
214  }
215  memcpy(apduPayloadToEncrypt, &(pApdu->pBuf[payloadOffset]), (pApdu->buflen - payloadOffset));
216  DPRINTF("SCP: Encrypting %d byte\r\n", pApdu->buflen - payloadOffset);
217 
218

221  U32 outLen = SCP_KEY_SIZE;
222 
223  memset(&mechInfo, 0, sizeof(mechInfo));
224  mechInfo.mechanism = HLSE_AES_CBC_ENCRYPT;
225  mechInfo.pParameter = pIv;
226  mechInfo.ulParameterLen = 16;
227

229  session.sEnc, SCP_KEY_SIZE,
230  apduPayloadToEncrypt, pApdu->buflen - payloadOffset,
231  &(pApdu->pBuf[payloadOffset]), &outLen);

233 // nRet = HOST_AES_CBC_Process(session.sEnc, SCP_KEY_SIZE, pIv, HOST_ENCRYPT, apduPayloadToEncrypt,

235  }
236  }

238  {

240  * The length of the command message (Lc) shall be incremented by 8 to indicate the inclusion of the
241  * C-MAC in the data field of the command message.
242  */
243  Lcc = pApdu->lc + CMAC_SIZE;
244  // The MAC will become the payload of the APDU. so indicate there is a datapayload
245  pApdu->hasData = 1;
246  pApdu->lcLength = 1;
247  pApdu->buflen += 1;

263 /*
264  * scp_PadData
265  *
266  * (spec 4.1.4) Unless specified otherwise, padding prior to performing an AES operation across a block of data is
267  * achieved in the following manner:
268  * - Append an '80' to the right of the data block;
269  * - If the resultant data block length is a multiple of 16, no further padding is required;
270  * - Append binary zeroes to the right of the data block until the data block length is a multiple of 16.
271  */
272 static U16 scp_PadData(apdu_t * pApdu)
273 {
274  U16 zeroBytesToPad = 0;
275  U16 bytesToPad = 1;
276 
277  // pad the payload and adjust the length of the APDU
278  if (pApdu->extendedLength == false)
279  {
280  // TODO: remove hard coded values
281  if (pApdu->buflen > 5) // payload present => padding needed
282  {
283  pApdu->pBuf[pApdu->buflen++] = 0x80;
284 
285  zeroBytesToPad = (SCP_KEY_SIZE - ((pApdu->buflen - 5) % SCP_KEY_SIZE)) % SCP_KEY_SIZE;
286  }
287  }
288  else
289  {
290  if (pApdu->buflen > 7) // payload present => padding needed
291  {
292  pApdu->pBuf[pApdu->buflen++] = 0x80;
293  zeroBytesToPad = (SCP_KEY_SIZE - ((pApdu->buflen - 7) % SCP_KEY_SIZE)) % SCP_KEY_SIZE;
294  }
295  }
296 
297  bytesToPad += zeroBytesToPad;
298 
299  while (zeroBytesToPad > 0)
300  {
301  pApdu->pBuf[pApdu->buflen++] = 0x00;
302  zeroBytesToPad--;
303  }
304 
305  pApdu->offset = pApdu->buflen;
306 
307  // DPRINTF("Padded %d bytes.\r\n", bytesToPad);
308  return bytesToPad;
309 }

432

434  {

436  {
437  i--;
438  }
