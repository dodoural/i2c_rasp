20 extern volatile uint32_t gtimer_kinetis_msticks; // counter for 1ms SysTicks
21 
22 //__INLINE
23 static void systick_delay(const uint32_t delayTicks) {
24  uint32_t currentTicks;
25  assert(delayTicks < 0x7FFFFFFFu);
26 
27  __disable_irq();
28 
29  if ((gtimer_kinetis_msticks) & 0x80000000u)
30  {
31  /* gtimer_kinetis_msticks has increased drastically (MSB is set),
32  * So, reset gtimer_kinetis_msticks before it's too late to detect an
33  * overflow. */
34  gtimer_kinetis_msticks = 0;
35  }
36 
37  currentTicks = gtimer_kinetis_msticks; // read current tick counter
38 
39  __DSB();
40  __enable_irq();
41 
42  // Now loop until required number of ticks passes
43  while ((gtimer_kinetis_msticks - currentTicks) <= delayTicks) {
44 #if !defined(IMX_RT)
45 #ifdef __WFI
46  __WFI();
47 #endif
48 #endif
49  }
50 }
51 
52 /* interrupt handler for system ticks */
53 void SysTick_Handler(void) {
54  gtimer_kinetis_msticks++;
55 }

58 /* initializes the system tick counter
59  * return 0 on succes, 1 on failure */
60 uint32_t sm_initSleep() {
61  gtimer_kinetis_msticks = 0;
62  SysTick_Config(SystemCoreClock / 1000);
63  __enable_irq();
64  return 0;
65 }

70 void sm_sleep(uint32_t msec) {
71  systick_delay(msec);
72 }
