29 # execCommand will stop script execution when the program executed did not return OK (i.e. 0) to the shell
30 execCommand () {
31  local command="$*"
32  echo ">> ${command}"
33  ${command}
34  local nRetProc="$?"
35  if [ ${nRetProc} -ne 0 ]
36  then
37  echo "\"${command}\" failed to run successfully, returned ${nRetProc}"
38  exit 2
39  fi
40  echo ""
41 }
