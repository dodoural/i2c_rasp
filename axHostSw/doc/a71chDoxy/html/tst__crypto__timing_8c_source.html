484 static U8 tstCryptoTimingProvision()
485 {
486  U8 result = 1;
487  U16 err = 0;
488 
489  ECCCurve_t eccCurve = ECCCurve_NIST_P256;
490 
491  int indexAesKey = 0;
492 
493  const U16 expectedPubKeyLen = 65;
494  const U16 expectedPrivKeyLen = 32;
495 
496  SST_Index_t index;
497 
498  U8 uid[A71CH_MODULE_UNIQUE_ID_LEN];
499  U16 uidLen = sizeof(uid);
500 
501  PRINTF( "\r\n-----------\r\nStart tstCryptoTimingProvision()\r\n------------\r\n");
502 
503  // In a typical provisioning scenario the keypair matching the public key
504  // to be inserted would be created and stored securely on a back-end system.
505  // It's also possible to create the keypairs to be contained in the A71CH externally
506  // and to insert both private and public key during the provisioning phase. In this
507  // example the keypairs are generated on the A71CH itself and the private key never
508  // leaves the A71CH.
509  err = HOSTCRYPTO_GenerateEccKey(eccCurve, &eccKeyRootCA_0);
510  result &= AX_CHECK_SW(err, SW_OK, "err");
511  err = HOSTCRYPTO_GenerateEccKey(eccCurve, &eccKeyRootCA_1);
512  result &= AX_CHECK_SW(err, SW_OK, "err");
513 
514  err = HOSTCRYPTO_GetPublicKey(eccKeyRootCA_0, eccKcRootCA_0.pub, &(eccKcRootCA_0.pubLen), (64 << 1)+1);
515  result &= AX_CHECK_SW(err, SW_OK, "err");
516  err = HOSTCRYPTO_GetPrivateKey(eccKeyRootCA_0, eccKcRootCA_0.priv, &(eccKcRootCA_0.privLen), 64);
517  result &= AX_CHECK_SW(err, SW_OK, "err");
518  eccKcRootCA_0.bits = expectedPrivKeyLen << 3;
519  eccKcRootCA_0.curve = eccCurve;
520 
521  err = HOSTCRYPTO_GetPublicKey(eccKeyRootCA_1, eccKcRootCA_1.pub, &(eccKcRootCA_1.pubLen), (64 << 1)+1);
522  result &= AX_CHECK_SW(err, SW_OK, "err");
523  err = HOSTCRYPTO_GetPrivateKey(eccKeyRootCA_1, eccKcRootCA_1.priv, &(eccKcRootCA_1.privLen), 64);
524  result &= AX_CHECK_SW(err, SW_OK, "err");
525  eccKcRootCA_1.bits = expectedPrivKeyLen << 3;
526  eccKcRootCA_1.curve = eccCurve;
527 
528  // Generate the first ECC keyPair (Extract the public key)
529  index = A71CH_KEY_PAIR_0;
530  PRINTF("\r\nA71_GenerateEccKeyPair(0x%02x)\r\n", index);
531  err = A71_GenerateEccKeyPair(index);
532  result &= AX_CHECK_SW(err, SW_OK, "err");
533 
534  // Generate the second ECC keyPair (Extract the public key)
535  index = A71CH_KEY_PAIR_1;
536  PRINTF("\r\nA71_GenerateEccKeyPair(0x%02x)\r\n", index);
537  err = A71_GenerateEccKeyPair(index);
538  result &= AX_CHECK_SW(err, SW_OK, "err");
539 
540  // Both ECC Key pairs have been created, now extract the value of the public key
541  index = A71CH_KEY_PAIR_0;
542  PRINTF( "\r\nSST_GetPublicKeyECCKeyPair(0x%02x)\r\n", index);
543  eccKcTls_0.pubLen = sizeof(eccKcTls_0.pub);
544  err = A71_GetPublicKeyEccKeyPair(index, eccKcTls_0.pub, &(eccKcTls_0.pubLen));
545  result &= AX_CHECK_SW(err, SW_OK, "err");
546  result &= AX_CHECK_U16(eccKcTls_0.pubLen, expectedPubKeyLen, "Wrong public key length");
547  // Complete eccKcTls_0 datastructure
548  if (err == SW_OK)
549  {
550  eccKcTls_0.bits = 256;
551  eccKcTls_0.curve = eccCurve;
552  eccKcTls_0.privLen = 0;
553  }
554 
555  index = A71CH_KEY_PAIR_1;
556  PRINTF( "\r\nSST_GetPublicKeyECCKeyPair(0x%02x)\r\n", index);
557  eccKcTls_1.pubLen = sizeof(eccKcTls_1.pub);
558  err = A71_GetPublicKeyEccKeyPair(index, eccKcTls_1.pub, &(eccKcTls_1.pubLen));
559  result &= AX_CHECK_SW(err, SW_OK, "err");
560  result &= AX_CHECK_U16(eccKcTls_1.pubLen, expectedPubKeyLen, "Wrong public key length");
561  // Complete eccKcTls_1 datastructure
562  if (err == SW_OK)
563  {
564  eccKcTls_1.bits = 256;
565  eccKcTls_1.curve = eccCurve;
566  eccKcTls_1.privLen = 0;
567  }
568 
569  // Set the public key of eccKcRootCA_0 on the A71CH (index=0)
570  index = A71CH_PUBLIC_KEY_0;
571  PRINTF("\r\nA71_SetEccPublicKey(0x%02x)\r\n", index);
572  err = A71_SetEccPublicKey(index, eccKcRootCA_0.pub, eccKcRootCA_0.pubLen);
573  result &= AX_CHECK_SW(err, SW_OK, "err");
574 
575  // Set the public key of eccKcRootCA_1 on the A71CH (index=1)
576  index = A71CH_PUBLIC_KEY_1;
577  PRINTF("\r\nA71_SetEccPublicKey(0x%02x)\r\n", index);
578  err = A71_SetEccPublicKey(index, eccKcRootCA_1.pub, eccKcRootCA_1.pubLen);
579  result &= AX_CHECK_SW(err, SW_OK, "err");
580 
581  // Fill the SYM key store with A71CH_SYM_KEY_MAX keys (128 bits long)
582  // The keys being written are based upon a Base pattern, with the first key written having a 0x00 byte
583  // in position 0, the second key written a 0x00 byte in position 1 and so on
584  // Base Pattern: 0xF0F1F2F3F4F5F6F7F8F9FAFBFCFDFEFF
585  // First Key : 0x00F1F2F3F4F5F6F7F8F9FAFBFCFDFEFF
586  // Second Key : 0xF000F2F3F4F5F6F7F8F9FAFBFCFDFEFF
587  // ..
588  for (indexAesKey=0; indexAesKey<A71CH_SYM_KEY_MAX; indexAesKey++)
589  {
590  memcpy(aesRef[indexAesKey], aesBasePattern, aesBasePatternLen);
591  // Put the 0x00 marker
592  aesRef[indexAesKey][indexAesKey] = (U8)0x00;
593  // Write the key (unwrapped)
594  PRINTF( "\r\nA71_SetSymKey(0x%02x)\r\n", indexAesKey);
595  err = A71_SetSymKey((SST_Index_t)indexAesKey, aesRef[indexAesKey], sizeof(aesRef[indexAesKey]));
596  result &= AX_CHECK_SW(err, SW_OK, "err");
597  axPrintByteArray("aesRef[indexAesKey]", aesRef[indexAesKey], sizeof(aesRef[indexAesKey]), AX_COLON_32);
598  }
599 
600  // Write end-product specific data into GP storage.
601  // This data can be end-product trim data, configuration data, ...
602  // The host or the backend can encrypt with a key derived from a a key in the SYM key store before
603  // storing the data in GP storage.
604  err = A71_SetGpData(productDataOffset, gpProductData, (U16) sizeof(gpProductData));
605  result &= AX_CHECK_SW(err, SW_OK, "err");
606 
607  // Fetch UID from A71CH
608  err = A71_GetUniqueID(uid, &uidLen);
609  result &= AX_CHECK_SW(err, SW_OK, "err");
610 
611  // Possible additional steps
612  // - Create Certificate Signing Request
613  // - Store a certificate (or the hash of a certificate) in GP storage
614  // - Store public keys in GP storage
615 
616  PRINTF( "\r\n-----------\r\nEnd tstCryptoTimingProvision(), result = %s\r\n------------\r\n",
617  ((result == 1)? "OK": "FAILED"));
618 
619  return result;
620 }
