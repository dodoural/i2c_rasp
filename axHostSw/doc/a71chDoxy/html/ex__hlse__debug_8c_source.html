190  {
191  // Fill sndBuffer with monotonic rising/wrapping data
192  for (j=0; j<i; j++) { sndBuf[j] = (U8)j; }
193  for (j=0; j<i; j++) { rcvBuf[j] = (U8)0xFF; }
194  rcvBufLen = sizeof(rcvBuf);

204  result &= 0;
